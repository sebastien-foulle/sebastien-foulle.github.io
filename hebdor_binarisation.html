Introduction

Plusieurs packages permettent de binariser toutes les variables qualitatives d’une table, on teste ici recipes, caret et mlrCPO qui complete mlr.

library("recipes")
library("mlrCPO")
library("caret")
library("dplyr")
library("kableExtra")

# jeu de donnees du package reshape2
donnees = reshape2::tips

Avec une modalite de reference qui est enlevee

Attention a ne pas oublier l’operateur %>>% quand on utilise le package mlrCPO.

donnees %>% recipe %>% step_dummy(all_nominal()) %>% prep %>% bake(new_data = donnees) %>% 
  head(n = 3) %>% kable(caption = "recipes") %>% kable_styling(full_width = F, font_size = 12)

dummy_caret = caret::dummyVars(" ~ .", data = donnees,fullRank = TRUE)
data.frame(predict(dummy_caret, newdata = donnees)) %>% 
  head(n = 3) %>% kable(caption = "caret") %>% kable_styling(full_width = F, font_size = 12)

donnees %>>% cpoDummyEncode(reference.cat = TRUE) %>% 
  head(n = 3) %>% kable(caption = "mlrCPO") %>% kable_styling(full_width = F, font_size = 12)

Avec toutes les modalites ( = “one-hot encoding”)

donnees %>% recipe %>% step_dummy(all_nominal(), one_hot = TRUE) %>% prep %>% 
  bake(new_data = donnees) %>% head(n = 3) %>% kable(caption = "recipes") %>% 
  kable_styling(full_width = F, font_size = 12)

dummy_caret2 = caret::dummyVars(" ~ .", data = donnees,fullRank = FALSE)
data.frame(predict(dummy_caret2, newdata = donnees)) %>% 
  head(n = 3) %>% kable(caption = "caret") %>% kable_styling(full_width = F, font_size = 12)
